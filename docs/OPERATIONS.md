## üíæ Backup & Recovery

### Automated Backup Script

```bash
#!/bin/bash
# backup.sh - Automated backup script

BACKUP_DIR="/opt/ai-prompt-manager/backups"
DATE=$(date +%Y%m%d_%H%M%S)
RETENTION_DAYS=30

# Create backup directory
mkdir -p "$BACKUP_DIR"

# SQLite backup
if [ -f "prompts.db" ]; then
    echo "üì¶ Creating SQLite backup..."
    cp prompts.db "$BACKUP_DIR/prompts_$DATE.db"
    gzip "$BACKUP_DIR/prompts_$DATE.db"
    echo "‚úÖ SQLite backup created: prompts_$DATE.db.gz"
fi

# PostgreSQL backup
if docker-compose ps | grep -q "db.*Up"; then
    echo "üì¶ Creating PostgreSQL backup..."
    docker-compose exec -T db pg_dump -U promptman promptman > "$BACKUP_DIR/postgres_$DATE.sql"
    gzip "$BACKUP_DIR/postgres_$DATE.sql"
    echo "‚úÖ PostgreSQL backup created: postgres_$DATE.sql.gz"
fi

# Configuration backup
echo "üì¶ Creating configuration backup..."
tar -czf "$BACKUP_DIR/config_$DATE.tar.gz" .env docker-compose.*.yml logs/

# Cleanup old backups
echo "üßπ Cleaning up old backups..."
find "$BACKUP_DIR" -name "*.gz" -mtime +$RETENTION_DAYS -delete
find "$BACKUP_DIR" -name "*.tar.gz" -mtime +$RETENTION_DAYS -delete

echo "‚úÖ Backup completed successfully!"
```

# monitor.sh - System monitoring script

# System metrics

MEM_USAGE=$(free | grep Mem | awk '{printf "%.2f", $3/$2 * 100.0}')

# Application metrics
WEB_RESPONSE=$(curl -o /dev/null -s -w "%{time_total}" http://localhost:7860/health)
API_RESPONSE=$(curl -o /dev/null -s -w "%{time_total}" http://localhost:7861/api/health)

# Database metrics
if [ -f "prompts.db" ]; then
    DB_SIZE=$(ls -lh prompts.db | awk '{print $5}')
else
    DB_SIZE="N/A"
fi

# Log metrics
echo "$DATE,CPU:$CPU_USAGE%,MEM:$MEM_USAGE%,DISK:$DISK_USAGE%,WEB:${WEB_RESPONSE}s,API:${API_RESPONSE}s,DB:$DB_SIZE" >> "$LOG_FILE"

# Alert on high usage
if (( $(echo "$CPU_USAGE > 80" | bc -l) )); then
    echo "‚ö†Ô∏è High CPU usage: $CPU_USAGE%"
fi

if (( $(echo "$MEM_USAGE > 80" | bc -l) )); then
    echo "‚ö†Ô∏è High memory usage: $MEM_USAGE%"
fi

### Security Hardening Checklist

```bash
# Security hardening script
#!/bin/bash

echo "üîí Security Hardening Checklist"
echo "==============================="

# 1. Update environment file permissions
if [ -f .env ]; then
    chmod 600 .env
    chown root:root .env
    echo "‚úÖ Environment file permissions hardened"
else
    echo "‚ö†Ô∏è Environment file not found"
fi

# 2. Update database permissions
if [ -f prompts.db ]; then
    chmod 640 prompts.db
    echo "‚úÖ Database file permissions hardened"
fi

# 3. Set up log rotation
cat > /etc/logrotate.d/ai-prompt-manager << EOF
/opt/ai-prompt-manager/logs/*.log {
    daily
    missingok
    rotate 30
    compress
    delaycompress
    notifempty
    sharedscripts
    postrotate
        docker-compose -f /opt/ai-prompt-manager/docker-compose.prod.yml restart app
    endscript
}
EOF
echo "‚úÖ Log rotation configured"

# 4. Set up firewall rules
if command -v ufw &> /dev/null; then
    ufw --force reset
    ufw default deny incoming
    ufw default allow outgoing
    ufw allow ssh
    ufw allow 80/tcp
    ufw allow 443/tcp
    ufw --force enable
    echo "‚úÖ Firewall configured"
else
    echo "‚ö†Ô∏è UFW not available - configure firewall manually"
fi

# 5. Disable unnecessary services
systemctl disable --now telnet
systemctl disable --now rsh
systemctl disable --now rlogin
echo "‚úÖ Unnecessary services disabled"

echo ""
echo "‚úÖ Security hardening completed"
```

### Daily Maintenance Script

```bash
#!/bin/bash
# daily-maintenance.sh

LOG_FILE="/opt/ai-prompt-manager/logs/maintenance.log"
DATE=$(date '+%Y-%m-%d %H:%M:%S')

echo "[$DATE] Starting daily maintenance" >> "$LOG_FILE"

# 1. Health check
if curl -f -s http://localhost:7860/health > /dev/null; then
    echo "[$DATE] ‚úÖ Health check passed" >> "$LOG_FILE"
else
    echo "[$DATE] ‚ùå Health check failed" >> "$LOG_FILE"
    # Restart service
    docker-compose restart app
    echo "[$DATE] üîÑ Service restarted" >> "$LOG_FILE"
fi

# 2. Log cleanup
find logs/ -name "*.log" -size +100M -exec truncate -s 50M {} \;
echo "[$DATE] ‚úÖ Log files cleaned" >> "$LOG_FILE"

# 3. Database optimization
if [ -f prompts.db ]; then
    sqlite3 prompts.db "VACUUM; PRAGMA optimize;"
    echo "[$DATE] ‚úÖ SQLite database optimized" >> "$LOG_FILE"
fi

# 4. Docker cleanup
docker system prune -f > /dev/null 2>&1
echo "[$DATE] ‚úÖ Docker cleanup completed" >> "$LOG_FILE"

# 5. Backup verification
LATEST_BACKUP=$(ls -t backups/*.gz 2>/dev/null | head -1)
if [ -n "$LATEST_BACKUP" ]; then
    BACKUP_AGE=$(( $(date +%s) - $(stat -c %Y "$LATEST_BACKUP") ))
    if [ $BACKUP_AGE -lt 86400 ]; then  # 24 hours
        echo "[$DATE] ‚úÖ Recent backup found" >> "$LOG_FILE"
    else
        echo "[$DATE] ‚ö†Ô∏è No recent backup found" >> "$LOG_FILE"
        ./backup.sh
    fi
else
    echo "[$DATE] ‚ö†Ô∏è No backups found" >> "$LOG_FILE"
    ./backup.sh
fi

echo "[$DATE] Daily maintenance completed" >> "$LOG_FILE"
```
